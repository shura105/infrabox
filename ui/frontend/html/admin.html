🔑 SSH ключі

📋 Як згенерувати ключ на хості 
# 1. Підключитись до хосту і згенерувати ключ:
ssh-keygen -t ed25519 -f ~/.ssh/infrabox_id -C "infrabox-adm" -N ""

# 2. Дозволити вхід за цим ключем:
cat ~/.ssh/infrabox_id.pub >> ~/.ssh/authorized_keys
chmod 600 ~/.ssh/authorized_keys

# 3. Вивести приватний ключ (вставити в поле нижче):
cat ~/.ssh/infrabox_id
